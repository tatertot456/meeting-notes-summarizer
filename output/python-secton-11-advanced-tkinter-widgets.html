Radiobuttons

▶

Use radiobuttons for mutually exclusive choices — only one can be selected at a time. Use a StringVar (or IntVar) to track which button is selected across the group.

import tkinter as tk
from tkinter import ttk, StringVar, Radiobutton

window = tk.Tk()

# StringVar — a window-based variable for holding widget state
v = StringVar(window, "1")   # default selection = "1"

# Dictionary makes it easy to create multiple buttons
values = {
    "RadioButton 1": "1",
    "RadioButton 2": "2",
    "RadioButton 3": "3",
}

# Create all buttons in a loop
for text, value in values.items():
    Radiobutton(
        window, text=text, variable=v,
        value=value, command=lambda: print(v.get())
    ).pack(side="top", ipady=5)

# Styled as actual buttons (indicator=0 removes the circle dot)
for text, value in values.items():
    Radiobutton(
        window, text=text, variable=v, value=value,
        indicator=0, background="light blue"
    ).pack(fill="x", ipady=5)

# Event binding — use ButtonRelease-1 not Button-1
# because the variable updates AFTER the click event
for text, value in values.items():
    rb = Radiobutton(window, text=text, variable=v, value=value)
    rb.bind("<ButtonRelease-1>", lambda e: print(v.get()))
    rb.pack(side="top", ipady=5)

# Trace — attach callback to variable change (most reliable approach)
v.trace("w", lambda name, index, mode, var=v: print(var.get()))

window.mainloop()

The key gotcha: binding to Button-1 (mouse down) fires BEFORE the StringVar updates — so you read the OLD value. Use ButtonRelease-1 (mouse up) or trace() to reliably read the NEW value after selection. trace("w", callback) fires whenever the variable is written to — this is the most reliable approach.

Checkbuttons

▶

Use checkbuttons for non-exclusive choices — multiple can be selected. Also great as a toggle (on/off switch). Each has an onvalue and offvalue.

import tkinter as tk
window = tk.Tk()

# IntVar to track checked state (1=checked, 0=unchecked)
var1 = tk.IntVar()
var2 = tk.IntVar()

checkbox1 = tk.Checkbutton(
    window,
    text="Tutorial",
    variable=var1,
    onvalue=1,     # value when checked
    offvalue=0,    # value when unchecked
    height=2,
    width=10
)
checkbox1.pack()

checkbox2 = tk.Checkbutton(window, text="Project", variable=var2)
checkbox2.pack()

def show_selections():
    print(f"Tutorial: {var1.get()}, Project: {var2.get()}")

tk.Button(window, text="Show", command=show_selections).pack()
window.mainloop()

Combobox

▶

A combobox is a dropdown list — more space-efficient than a listbox. The user can select from the list or type their own value. Requires ttk (themed tkinter).

import tkinter as tk
from tkinter import ttk

window = tk.Tk()

# StringVar to hold the selected value
n = tk.StringVar()

dayChosen = ttk.Combobox(window, width=27, textvariable=n)

# Set the dropdown options
dayChosen["values"] = (
    "Monday", "Tuesday", "Wednesday",
    "Thursday", "Friday", "Saturday", "Sunday"
)
dayChosen.pack()

# Get the selected value
def show_choice():
    print(n.get())          # what's in the StringVar
    print(dayChosen.get())  # same thing, via widget

tk.Button(window, text="Show", command=show_choice).pack()
window.mainloop()

Menus — Menubar, Popup & Menubutton

▶

Menus sit on the menubar at the top of the window and handle app-wide functions (File, Edit, Help). Build them in steps: menubar → cascade → commands.

from tkinter import Menu, Menubutton # Step 1: Create menubar menubar = Menu(window) # Step 2: Create top-level "File" menu file = Menu(menubar, tearoff=0) # tearoff=0 prevents detaching # Step 3: Add commands to File menu file.add_command(label="New File", command=mnuNewFile) file.add_command(label="Open File", command=mnuOpenFile) file.add_command(label="Save File", command=mnuSaveFile) file.add_separator() # visual divider file.add_command(label="Exit", command=window.destroy) # Step 4: Attach File menu to menubar menubar.add_cascade(label="File", menu=file) # Step 5: Attach menubar to window window.config(menu=menubar) # POPUP MENU — right-click context menu popup_menu = Menu(window, tearoff=0) popup_menu.add_command(label="say hi", command=lambda: hey("hi")) popup_menu.add_command(label="say hello", command=lambda: hey("hello")) popup_menu.add_separator() popup_menu.add_command(label="say bye", command=lambda: hey("bye")) # Bind right-click to show popup window.bind("<Button-3>", lambda e: popup_menu.post(e.x_root, e.y_root)) # MENUBUTTON — free-floating menu (not on menubar) var1, var2, var3 = tk.IntVar(), tk.IntVar(), tk.IntVar() menubutton = Menubutton(window, text="Options") menubutton.menu = Menu(menubutton) menubutton["menu"] = menubutton.menu menubutton.menu.add_checkbutton(label="Courses", variable=var1) menubutton.menu.add_checkbutton(label="Students", variable=var2) menubutton.menu.add_checkbutton(label="Careers", variable=var3) menubutton.pack()

Lambdas in menus: command= expects a function with NO arguments. To pass arguments, wrap with a lambda: command=lambda: hey("hi"). This creates an anonymous function that calls hey("hi") when the menu item is selected — without the lambda you'd have to create a separate named function for each menu item.

Listbox

▶

A listbox displays a scrollable list of selectable items — like a visible array. Supports single or multiple selection.

import tkinter as tk
from tkinter.messagebox import showinfo

window = tk.Tk()

# Method 1: insert items manually
listbox = tk.Listbox(
    window, height=10, width=15,
    bg="grey", activestyle="dotbox",
    font="Helvetica", fg="yellow"
)
listbox.insert(1, "Freshman")
listbox.insert(2, "Sophomore")
listbox.insert(3, "Junior")
listbox.insert(4, "Senior")
listbox.pack()

# Method 2: use a Variable object (cleaner)
langs = ("Java", "C#", "C", "C++", "Python", "Go", "JavaScript")
varList = tk.Variable(value=langs)
listbox2 = tk.Listbox(
    window,
    listvariable=varList,
    height=6,
    selectmode=tk.EXTENDED   # allows multiple selections
)
listbox2.pack()

# Get selected items
for i in listbox.curselection():
    print(listbox.get(i))

# Event handler for selection change
def items_selected(event):
    selected_indices = listbox2.curselection()
    selected_langs = ", ".join([listbox2.get(i) for i in selected_indices])
    showinfo(title="Selection", message=f"You selected: {selected_langs}")

listbox2.bind("<<ListboxSelect>>", items_selected)
window.mainloop()

curselection() returns a tuple of selected indices. Use listbox.get(i) to retrieve the value at each index. For multiple selection, iterate through all indices. selectmode=tk.EXTENDED allows Shift+click and Ctrl+click for multiple selections. selectmode=tk.SINGLE restricts to one item at a time.

Spinbox & Progress Bar

▶
from tkinter import Spinbox, Progressbar, HORIZONTAL # SPINBOX — numeric input with min/max limits and up/down arrows sp = Spinbox(window, from_=0, to=20) sp.pack() # Get current value print(sp.get()) # PROGRESS BAR — shows completion status from tkinter.ttk import Progressbar progress = Progressbar( window, orient=HORIZONTAL, length=100, mode="determinate" # shows actual % progress # mode="indeterminate" — bouncing bar for unknown duration ) progress.pack() def ShowBar(): bar_value = 0 while bar_value <= 100: progress["value"] = bar_value window.update_idletasks() # force redraw — REQUIRED bar_value += 10 window.after(100) # pause 100ms between steps tk.Button(window, text="Start", command=ShowBar).pack() window.mainloop()

update_idletasks() is critical for the progress bar — without it the window freezes during the loop and you never see the bar move. It forces tkinter to process any pending drawing events. For long-running tasks, consider using threading instead of a while loop to keep the UI responsive.

Additional Windows & Images

▶
from tkinter import Toplevel, Label from PIL import Image, ImageTk # TOPLEVEL — open a second window def open_new_window(): newWindow = Toplevel(window) newWindow.title("New Window") newWindow.geometry("200x200") Label(newWindow, text="This is a new window!").pack() tk.Button(window, text="Open Window", command=open_new_window).pack() # IMAGES — use PIL/Pillow library for image support img = ImageTk.PhotoImage(Image.open("../images/photo.jpg")) panel = tk.Label(window, image=img) panel.pack() # IMAGE ON BUTTON photoimage = ImageTk.PhotoImage(Image.open("icon.png")) tk.Button( window, text="Click Me!", image=photoimage, compound="left" # text to right of image ).pack(side="top") # CUSTOM APP ICON window.iconbitmap(r"C:\path\to\icon.ico") window.title("Custom Icon App")

Standard Dialogs

▶

Tkinter provides built-in standard dialogs so you don't have to build them from scratch — file open/save, color chooser, and more.

from tkinter.filedialog import askopenfile, asksaveasfile from tkinter import colorchooser # FILE OPEN DIALOG def open_file(): file = askopenfile( mode="r", filetypes=[("Python Files", "*.py"), ("All Files", "*.*")] ) if file is not None: content = file.read() print(content) file.close() # FILE SAVE DIALOG def save_file(): file_types = [ ("All Files", "*.*"), ("Python Files", "*.py"), ("Text Document", "*.txt") ] file = asksaveasfile( filetypes=file_types, defaultextension=file_types ) if file is not None: file.write("Content to save") file.close() # COLOR CHOOSER DIALOG def choose_color(): color_code = colorchooser.askcolor(title="Choose color") print(color_code) # Returns: ((R, G, B), "#hexcode") or (None, None) if cancelled tk.Button(window, text="Open File", command=open_file).pack() tk.Button(window, text="Save File", command=save_file).pack() tk.Button(window, text="Pick Color", command=choose_color).pack() window.mainloop()

Always check if the returned file is not None before using it — the user can cancel the dialog, which returns None. The color chooser returns a tuple: the first element is an (R, G, B) tuple, the second is the hex string "#rrggbb". Both are None if the user cancels. These dialogs are native OS dialogs — they match the user's operating system look and feel automatically.
